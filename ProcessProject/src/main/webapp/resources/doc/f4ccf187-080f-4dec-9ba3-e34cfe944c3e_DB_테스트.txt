SELECT  EMPLOYEE_ID,
        FIRST_NAME || ' ' || LAST_NAME names,
        DECODE(ROUND((SYSDATE - HIRE_DATE)/365),   5, '5년근속',
        10, '10년근속',
        15, '15년근속',
        20, '20년근속',
               ROUND((SYSDATE - HIRE_DATE)/365)) work_years
FROM    EMPLOYEES
ORDER BY EMPLOYEE_ID;



SELECT  EMPLOYEE_ID,
        ROUND((SYSDATE - HIRE_DATE)/365) w_years,
        CASE ROUND((SYSDATE - HIRE_DATE)/365) WHEN 5 THEN '5년근속'
          WHEN 10 THEN '10년근속'
          WHEN 15 THEN '15년근속'
          WHEN 20 THEN '20년근속'
          ELSE to_char(ROUND((SYSDATE - HIRE_DATE)/365))
        END work_years
FROM    EMPLOYEES;
